% ---------------------------------------------------------------- 1. problema
\begin{frame}{Problema e aplicação}
  \begin{itemize}
    \item \textbf{Multiplicação de matrizes densas} $C = A \times B$, dupla precisão, $N = 1024$.
    \item $2 N^3 \approx 2{,}15$ GFLOP por execução; matrizes de 8 MiB (24 MiB no total).
    \item Núcleo de álgebra linear, simulação e aprendizado de máquina; caso clássico de kernel com \textbf{reuso de dados} e \textbf{paralelismo abundante}.
  \end{itemize}
  \begin{block}{Pergunta da investigação}
    Dado um código sequencial com gargalo conhecido, \emph{quais intervenções OpenMP produzem ganho real, quais não produzem, e por quê?}
  \end{block}
  \begin{itemize}
    \item O objetivo \textbf{não} é competir com BLAS, e sim explicar a relação entre arquitetura, técnica e resultado.
  \end{itemize}
\end{frame}

% ---------------------------------------------------------------- 3. baseline
\begin{frame}[fragile]{Baseline e metodologia de medição}
  \begin{columns}[T]
    \column{0.46\textwidth}
      \textbf{v0: laços i-j-k canônicos}
\begin{semiverbatim}\small
for (i = 0; i < n; i++)
  for (j = 0; j < n; j++) \{
    double sum = 0.0;
    for (k = 0; k < n; k++)
      sum += A[i*n+k] * B[k*n+j];
    C[i*n+j] = sum;
  \}
\end{semiverbatim}
      \textbf{Baseline: 2,298 s} ($\sigma = 0{,}0003$) $\rightarrow$ 0,94 GFLOP/s
    \column{0.50\textwidth}
      \textbf{Metodologia}
      \begin{itemize}
        \item 1 warm-up não medido + \textbf{5 repetições}; mediana $\pm$ desvio ($< 1\%$);
        \item threads fixadas nos cores físicos\\ (\texttt{OMP\_PROC\_BIND=close}, \texttt{OMP\_PLACES=cores});
        \item mesmas entradas em todas as versões (gerador determinístico);
        \item \textbf{validação}: comparação elemento a elemento com a v0 --- diferença \textbf{zero} nas 4 versões.
      \end{itemize}
  \end{columns}
\end{frame}

% ---------------------------------------------------------------- 4. diagnostico
\begin{frame}{Diagnóstico do gargalo}
  \begin{itemize}
    \item Pico teórico do nó $\approx$ 736 GFLOP/s (20 cores $\times$ 16 flops/ciclo $\times$ 2,3 GHz).
    \item A v0 atinge \textbf{0,94 GFLOP/s $\approx$ 0,1\% do pico do nó} $\rightarrow$ o gargalo \textbf{não} é capacidade de cômputo.
    \item Laço interno lê \texttt{B[k*n+j]} com \textbf{stride de $N$ elementos (8 KiB)}: uma linha de cache nova por iteração, usando 1 dos 8 doubles trazidos.
    \item Sintoma do mapa de decisão da disciplina: \emph{tempo dominado por memória / acessos com stride}.
  \end{itemize}
  \begin{block}{Hipóteses de intervenção}
    \begin{enumerate}
      \item[H1.] Vetorizar o produto escalar (\texttt{omp simd}) aproveita as unidades SIMD. \hfill \alert{(falsa)}
      \item[H2.] Paralelizar o laço externo com granularidade grossa escala com os cores. \hfill (verdadeira)
      \item[H3.] Trocar a ordem dos laços elimina o stride --- ataca a causa raiz. \hfill (verdadeira)
    \end{enumerate}
  \end{block}
\end{frame}

% ---------------------------------------------------------------- 5. otimizacoes v1
\begin{frame}[fragile]{Otimização v1 --- \texttt{omp simd} no laço interno (sem ganho)}
  \begin{columns}[T]
    \column{0.5\textwidth}
\begin{semiverbatim}\small
double sum = 0.0;
#pragma omp simd reduction(+:sum)
for (k = 0; k < n; k++)
  sum += A[i*n+k] * B[k*n+j];
\end{semiverbatim}
      \begin{itemize}
        \item \textbf{Hipótese (H1):} SIMD $\approx$ 4 doubles/instrução (AVX2) $\rightarrow$ ganho esperado $\sim$4$\times$.
      \end{itemize}
    \column{0.46\textwidth}
      \begin{itemize}
        \item O compilador \textbf{vetoriza de fato} (\texttt{-fopt-info-vec}), mas monta os vetores de \texttt{B} com cargas escalares --- o acesso tem stride.
        \item O tráfego de memória por iteração \textbf{não muda}.
      \end{itemize}
      \begin{alertblock}{Resultado medido}
        2,388 s $\rightarrow$ \textbf{0,96$\times$} (leve piora)
      \end{alertblock}
  \end{columns}
\end{frame}

% ---------------------------------------------------------------- 6. otimizacoes v2 v3
\begin{frame}[fragile]{Otimizações v2 e v3 --- a MESMA técnica, granularidades opostas}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{v2: laço externo (grossa)}
\begin{semiverbatim}\small
#pragma omp parallel for \\
        schedule(static)
for (i = 0; i < n; i++)
  ... /* corpo igual ao da v0 */
\end{semiverbatim}
      \begin{itemize}
        \item 1 região paralela; $N/p$ linhas por thread, sem sincronização interna.
        \item Corpo idêntico à v0 $\rightarrow$ isola o efeito do paralelismo (v2@1 $=$ 2,294 s $\approx$ v0).
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{v3: laço interno (fina)}
\begin{semiverbatim}\small
for (i ...) for (j ...) \{
#pragma omp parallel for \\
        reduction(+:sum)
  for (k = 0; k < n; k++) ...
\}
\end{semiverbatim}
      \begin{itemize}
        \item Região paralela criada $N^2 \approx 10^6$ vezes;
        \item cada uma distribui só 1024 multiplicações e termina em reduction + \textbf{barreira entre 2 sockets}.
      \end{itemize}
  \end{columns}
\end{frame}

% ---------------------------------------------------------------- 7. otimizacao v4
\begin{frame}[fragile]{Otimização v4 --- troca de laços i-k-j (segunda iteração do ciclo)}
  \begin{columns}[T]
    \column{0.5\textwidth}
\begin{semiverbatim}\small
#pragma omp parallel for \\
        schedule(static)
for (i = 0; i < n; i++) \{
  for (j ...) C[i*n+j] = 0.0;
  for (k = 0; k < n; k++) \{
    const double a = A[i*n+k];
    for (j = 0; j < n; j++)
      C[i*n+j] += a * B[k*n+j];
  \}
\}
\end{semiverbatim}
    \column{0.46\textwidth}
      \begin{itemize}
        \item Após a v2 o ciclo volta ao diagnóstico: throughput ainda $\approx$ 2\% do pico $\rightarrow$ o stride continua lá.
        \item Laço interno passa a percorrer \texttt{B} e \texttt{C} com \textbf{stride 1}: vetorização efetiva + prefetch sequencial + linha de cache usada por inteiro.
        \item Soma na mesma ordem de $k$ $\rightarrow$ resultado \textbf{bit a bit idêntico} ao da v0.
      \end{itemize}
  \end{columns}
\end{frame}

% ---------------------------------------------------------------- 13. discussao
\begin{frame}{Discussão: por que cada resultado aconteceu}
  \begin{itemize}
    \item \textbf{v1 (0,96$\times$):} SIMD multiplica cômputo, mas o tempo é gasto esperando memória; vetorizar sem mudar o padrão de acesso não altera o tráfego. \emph{Uma técnica só é boa otimização quando resolve o gargalo dominante.}
    \item \textbf{v3 (0,23$\times$):} $10^6$ regiões paralelas $\Rightarrow$ overhead de fork/join + reduction + barreiras \textbf{atravessando 2 sockets NUMA} custa mais que todo o trabalho útil. Granularidade fina destrói o ganho do paralelismo.
    \item \textbf{v2 (17,5$\times$, 87\%):} granularidade grossa escala quase linearmente; a queda suave de eficiência coincide com a ocupação do segundo socket (acessos remotos NUMA).
    \item \textbf{v4 (132$\times$):} o par v1/v4 fecha a prova --- as mesmas unidades SIMD rendem 0,96$\times$ com stride e 7,9$\times$ sequencial (1 thread) quando o acesso é corrigido; com 20 threads, 123,5 GFLOP/s ($\approx$ 17\% do pico).
  \end{itemize}
\end{frame}

% ---------------------------------------------------------------- 14. limitacoes
\begin{frame}{Limitações}
  \begin{itemize}
    \item Sem contadores de hardware (\texttt{perf}) no ambiente: o diagnóstico usa métricas derivadas (GFLOP/s, eficiência, escalabilidade) e análise do padrão de acesso.
    \item Matrizes inicializadas pela thread 0 (first-touch no socket 0): com $>10$ threads parte dos acessos é remota; afeta todas as versões igualmente, mas limita o teto de escalabilidade.
    \item A v4 usa $\approx$ 17\% do pico do nó: os próximos passos seriam tiling/blocking para registradores e L1, e $N$ maiores que estourem a L3.
    \item Tempos da v4 com 20 threads ($\approx$ 17 ms) são curtos; $N$ maior daria mais folga entre sinal e ruído.
  \end{itemize}
\end{frame}

% ---------------------------------------------------------------- 15. conclusao
\begin{frame}{Conclusão}
  \begin{block}{Melhor configuração}
    \textbf{v4} --- troca de laços i-k-j + \texttt{omp parallel for schedule(static)}, 20 threads fixadas nos cores físicos: \textbf{0,017 s, speedup de 132$\times$} sobre o baseline, com resultado bit a bit idêntico.
  \end{block}
  \begin{block}{Principal aprendizado técnico}
    \textbf{O valor de uma técnica depende do gargalo, não da técnica.} Sobre o mesmo kernel, construções OpenMP produziram 0,96$\times$ (SIMD sem mudar o tráfego), 0,23$\times$ (paralelismo com granularidade fina em 2 sockets), 17,5$\times$ (granularidade grossa) e 132$\times$ (paralelismo + correção do padrão de acesso apontado pelo diagnóstico).
  \end{block}
  \begin{itemize}
    \item Medição controlada (warm-up, mediana, pinning, governor, validação) foi o que permitiu distinguir e explicar os quatro desfechos.
  \end{itemize}
\end{frame}
